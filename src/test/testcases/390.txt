
        int  main()
        {
            int base, exponent;
            int result;
            result = 1;
            printf("Enter a base number: ");
            scanf("%d", base);
            printf("Enter an exponent: ");
            scanf("%d", exponent);
            do
            {
                result = result * base;
                exponent = exponent - 1;
            }while (exponent != 0);
            printf("Answer = %lld", result);
            return 0;
        }
